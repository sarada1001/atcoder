N,M = map(int,input().split())

counts = [0]*(N+1)
for _ in range(M):
  a,b = map(int,input().split())
  counts[a] += 1
  counts[b] += 1
  
ans = []

for i in range(1,N+1):
  c = N - 1 - counts[i]
  
  if c >=3:
    combinations = c*(c-1)*(c-2)//6
    ans.append(combinations)
    
  else:
    ans.append(0)


print(*ans)
